种获取进程路径的方式

#include<stdio.h>
#include<windows.h>
#include<Psapi.h>

int main()
{
    WCHAR szFilePath[MAX_PATH] = { 0 };
    //模块只能是本进程记载的，如何要找其他进程的模块，使用Ex版
    GetModuleFileName(LoadLibrary(L"user32.dll"), szFilePath, MAX_PATH);
    wprintf(L"%s\n", szFilePath);

    HANDLE hProcess = OpenProcess(PROCESS_ALL_ACCESS, FALSE, 6164);
    GetProcessImageFileName(hProcess, szFilePath, MAX_PATH);
    wprintf(L"%s\n", szFilePath);

    DWORD dwSize = 0;
    QueryFullProcessImageName(hProcess, 0, szFilePath, &dwSize);
    wprintf(L"%s\n", szFilePath);
}
